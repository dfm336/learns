public class Solution {
    public boolean hasCycle(ListNode head) {
        ListNode slow = head;
        ListNode quick = head;
        while (slow != null && quick != null){
            if (slow == quick){
                return true;
            }
            slow = slow.next;
            if (quick.next != null){
                quick = quick.next.next;
            }

        }
        return false;
    }
}
